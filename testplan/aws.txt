### Reset

## Config
ops config reset
ops cloud aws vm-list
#! error!
task aws:config
ops cloud aws vm-list
#! show list of vms (none)

## Create VM k3s
ops cloud aws vm-create k3s-test
ops cloud aws vm-create mk8s-test
ops cloud aws vm-list
#! see the vm 

# Install k3s
KIP=$(ops cloud aws vm-getip k3s-test)
ops cloud k3s create SERVER=$KIP USERNAME=ubuntu
ops config use

## Install microk8s
MIP=$(ops cloud aws vm-getip mk8s-test)
ops cloud mk8s create ERVER=$MIP USERNAME=ubuntu
ops config use

# recover config
ops config reset
task aws:config
ops config use
ops cloud k3s kubeconfig SERVER=$KIP USERNAME=ubuntu
ops config use
ops cloud mk8s kubeconfig SERVER=$MIP USERNAME=ubuntu
ops config use

# Delete
ops cloud mk8s delete SERVER=$MIP USERNAME=ubuntu
ops config use
ops cloud k3s delete SERVER=$KIP USERNAME=ubuntu
ops config use

## Delete VMs
ops cloud aws vm-delete k3s-test
ops cloud aws vm-delete mk8s-test
ops cloud aws vm-list
